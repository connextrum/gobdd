Arguments
Name Description

path - String!

Example

Query
query article(
  $publication: String!,
  $path: String!
) {
  article(
    publication: $publication,
    path: $path
  ) {
    id
    type
    title
    seoTitle
    seoDescription
    publishDate
    updateDate
    adTags
    homeSection {
      ...HomeSectionFragment
    }

      "storyline": [StorylineElement]
    }
  }
